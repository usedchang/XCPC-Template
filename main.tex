struct DSU{
    vector<int>f;
    vector<int>siz;
    DSU(){};
    DSU(int n){
        f.resize(n+1);
        iota(f.begin(),f.end(),0);
        siz.assign(n+1,1);
    }
    int find(int x){
        return x==f[x]?x:f[x]=find(f[x]);
    }
    bool merge(int x,int y){
        int fx=find(x),fy=find(y);
        if(fx==fy) return false;
        f[fx]=fy;
        siz[fy]+=siz[fx];
        return true;
    }
    bool same(int x,int y){
        return find(x)==find(y);
    }
};

struct DSU{
    DSU(){}
    vector<int>f,siz;
    vector<ll>dis;
    DSU(int n){
        dis.resize(n+1);
        f.resize(n+1);
        iota(f.begin(),f.end(),0);

    }

        if(fx==x) return x;
        int rt=find(f[x]);
        dis[x]+=dis[f[x]];

struct Seg{
    #define lc (p<<1)
    #define rc (p<<1|1)
    struct S{
        int l,r;
        ll val,laz;
    };
    Seg(){};
    Seg(int n){
        tre.resize(n<<2);
        build(0,n,1);
    };
    void pushup(int p){
        tre[p].val=tre[lc].val+tre[rc].val;
    }
    void pushdown(int p){
        if(tre[p].laz==0) return;
        tre[lc].laz+=tre[p].laz;
        tre[rc].laz+=tre[p].laz;
        tre[lc].val+=tre[p].laz;
        tre[rc].val+=tre[p].laz;
        tre[p].laz=0;
    }
    void build(int l,int r,int p){
        tre[p]={l,r,0,0};
        if(l==r){
            tre[p]=a[l];
            return;
        }
        int mid=l+r>>1;
        build(l,mid,lc);
        build(mid+1,r,rc);
        pushup(p);
    }
    void add(int l,int r,ll k,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            tre[p].val+=k*(tre[p].r-tre[p].l+1);
            tre[p].laz+=k;
            return;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        if(l<=mid) add(l,r,k,lc);
        if(r>mid) add(l,r,k,rc);
        pushup(p);
    }
    ll qry(int l,int r,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            return tre[p].val;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        ll ans=0;
        if(l<=mid) ans+=qry(l,r,lc);
        if(r>mid) ans+=qry(l,r,rc);
        return ans;
    }
};

    int n;cin>>n;

            lc=st.back();
            st.pop_back();

        st.emplace_back(i);

        if(L[u]) dfs(dfs,L[u],l,u-1);
        if(R[u]) dfs(dfs,R[u],u+1,r);
        (dp[u]+=(dp[L[u]]+1)*(dp[R[u]]+1)%mod)%=mod;
        if(l>1) (dp[u]+=dp[R[u]])%=mod;
        if(r<n) (dp[u]+=dp[L[u]])%=mod;   
    };
    dfs(dfs,st.front(),1,n);

            p=son[p][num];
            ++cnt[p];

            else if(islower(c)) num=c-'a'+26;
            else if(isupper(c)) num=c-'A';
            if(!son[p][num]) return 0;
            p=son[p][num];
        }
        return cnt[p];
    }
}trie;

\subsection{根号分治}


\begin{itemize}
    \item 第一行：两个整数 $n, q$（$1 \le n, q \le 10^5$），分别表示序列长度和询问次数。
    \item 第二行：$n$ 个整数 $A_1, A_2, \dots, A_n$（$1 \le A_i \le 10^9$），表示给定的整数序列。
    \item 接下来 $q$ 行：每行两个整数 $x, y$（$1 \le x, y \le 10^9$），表示一次询问。
\end{itemize}

\section*{输出格式}
输出 $q$ 行，每行一个整数，表示对应询问的答案。

\section*{简要分析}
给定序列 $A[1..n]$ 和询问 $(x,y)$，需统计满足 $i<j$ 且 $A[i]=x, A[j]=y$ 的数对数量。记 $pos[v]$ 为值 $v$ 在序列中出现的位置集合（升序排列），则答案可表示为：
$$
\sum_{p \in pos[x]} \left( |pos[y]| - \text{upper\_bound}(pos[y], p) \right)
$$
或等价表示为：
$$
\sum_{p \in pos[y]} \text{lower\_bound}(pos[x], p)
$$

\section*{核心划分}
设阈值 $B = \sqrt{n}$（取整），将所有值分为两类：
\begin{enumerate}
    \item \textbf{高频值（Heavy）}：满足 $|pos[v]| > B$ 的值，记高频值集合为 $H$；
    \item \textbf{低频值（Light）}：满足 $|pos[v]| \le B$ 的值。
\end{enumerate}

\section*{预处理阶段（离线预处理高频值对）}
对于任意两个高频值 $u, v \in H$，预计算数对 $(u,v)$ 的答案并存储在哈希表 $mp$ 中：
$$
mp[(u,v)] = \sum_{p \in pos[u]} \left( |pos[v]| - \text{upper\_bound}(pos[v], p) \right)
$$
预处理时间复杂度：$O(|H|^2 \cdot B)$，由于 $|H| \le \frac{n}{B}$，故总复杂度为 $O(nB)$。

\section*{查询阶段（分情况处理）}
对每个询问 $(x,y)$，分三类处理（结合记忆化避免重复计算）：
\begin{enumerate}
    \item 若 $(x,y)$ 已计算过（记忆化哈希表 $\text{memo}$ 存在），直接返回结果；
    \item 若 $x,y$ 均为高频值，直接调用预处理结果 $mp[(x,y)]$；
    \item 若至少一个为低频值，选择更小的集合遍历计算：
          \begin{itemize}
              \item 若 $|pos[x]| \le |pos[y]|$：遍历 $pos[x]$，对每个位置 $p$ 统计 $pos[y]$ 中大于 $p$ 的元素数量；
              \item 否则：遍历 $pos[y]$，对每个位置 $p$ 统计 $pos[x]$ 中小于 $p$ 的元素数量。
          \end{itemize}
          单次查询时间复杂度：$O(B)$。
\end{enumerate}

\section*{复杂度分析}
预处理复杂度：$O(n\sqrt{n})$；
单次查询复杂度：$O(\sqrt{n})$，总查询复杂度：$O(q\sqrt{n})$；
整体时间复杂度：$O((n+q)\sqrt{n})$，可通过 $10^5$ 规模的数据。

void solve(){
    int n,q;cin>>n>>q;
    vector<int>a(n+1);
    unordered_map< int,vector<int> >pos;//存储重数
    for(int i=1;i<=n;i++){
        cin>>a[i];
        pos[a[i]].push_back(i);
    }
    int siz=sqrt(n+0.5);
    vector<int>H;
    for(auto const&[x,y]:pos){
        if(y.size()>siz) H.emplace_back(x);
    }
    map<pair<int,int>,ll>mp;
    for(int i=0;i<H.size();i++){
        for(int j=0;j<H.size();j++){
            for(int &p:pos[H[i]]){
                mp[{H[i],H[j]}]+=pos[H[j]].end()-upper_bound(pos[H[j]].begin(),pos[H[j]].end(),p);
            }
        }
    }
    map<pair<int,int>,ll>memo;
    for(int i=1;i<=q;i++){
        int x,y;cin>>x>>y;
        if(memo.count({x,y})){
            cout<<memo[{x,y}]<<endl;
            continue;
        }
        if(pos[x].size()>siz&&pos[y].size()>siz) {
            cout<<(memo[{x,y}]=mp[{x,y}])<<endl;
        }
        else{
            ll ans=0;
            if(pos[x].size()<=pos[y].size()||pos[x].size()<=siz&&pos[y].size()<=siz){
                for(int &p:pos[x]){
                    ans+=pos[y].end()-upper_bound(pos[y].begin(),pos[y].end(),p);
                }
            }
            else{
                for(int &p:pos[y]){
                    ans+=lower_bound(pos[x].begin(),pos[x].end(),p)-pos[x].begin();
                }
            }
            cout<<(memo[{x,y}]=ans)<<endl;
        }
    }
}

\subsubsection{无向图三元环计数}
\section*{题面简述}

给定一个 $n$ 个点 $m$ 条边的简单无向图（无重边、无自环，不保证连通），求图中三元环的个数。

三元环指恰好包含三个点 $u, v, w$ 及三条边 $\langle u,v\rangle, \langle v,w\rangle, \langle w,u\rangle$ 的子图。两个三元环不同当且仅当存在一个点属于其中一个而不属于另一个。

数据范围：$1 \le n \le 10^5$，$1 \le m \le 2 \times 10^5$。

\section*{做法}

采用经典的\textbf{无向边定向 + 枚举}算法。

\begin{enumerate}
    \item \textbf{定向}：对每条无向边 $(u, v)$，令其从度数小的点指向度数大的点；若度数相同，则从编号小的点指向编号大的点。这样将无向图转化为一个 DAG，记有向边集为 $E'$。
    \item \textbf{枚举}：对每个点 $u$，先用标记数组标记 $u$ 的所有出邻居。然后对每条出边 $u \to v$，枚举 $v$ 的所有出邻居 $w$，若 $w$ 被 $u$ 标记过（即 $u \to w \in E'$），则 $(u, v, w)$ 构成一个三元环，答案加一。
\end{enumerate}

由于定向后每个三元环的三条边恰好形成一条有向路径 $u \to v \to w$（其中 $u \to w$ 也存在），因此每个三元环恰好被统计一次，无需去重。

\section*{复杂度分析}

\textbf{引理}：定向后每个点的出度 $d^+(u) \le \sqrt{2m}$。

\begin{proof}
对任意点 $u$，其每个出邻居 $v$ 都满足 $\deg(v) \ge \deg(u) \ge d^+(u)$。因此 $u$ 的 $d^+(u)$ 个出邻居的度数之和满足：
\[
    \sum_{v \in N^+(u)} \deg(v) \ge (d^+(u))^2
\]
又因为所有边的度数贡献总和不超过 $2m$，故 $(d^+(u))^2 \le 2m$，即 $d^+(u) \le \sqrt{2m}$。
\end{proof}

算法的核心操作量为：
\[
    T = \sum_{(u \to v) \in E'} d^+(v) \le \sum_{(u \to v) \in E'} \sqrt{2m} = |E'| \cdot \sqrt{2m} = m\sqrt{2m}
\]

因此总时间复杂度为 $O(m\sqrt{m})$。

对于本题 $m = 2 \times 10^5$，操作量约为 $2 \times 10^5 \times 632 \approx 1.26 \times 10^8$，可以通过。空间复杂度为 $O(n + m)$。

struct Info{
    ll sum,mx,mn;
    Info(ll sum,ll mx,ll mn):sum(sum),mx(mx),mn(mn){};
    Info():sum(0LL),mx(-INF),mn(INF){};
    friend Info operator+(const Info A,const Info B){
        return Info(A.sum+B.sum,max(A.mx,B.mx),min(A.mn,B.mn));
    }

struct Seg{
    struct S{
        int l,r;
        ll sum,mx,mn,laz;bool laz1;
    };
    #define lc (p<<1)
    #define rc (p<<1|1)
    vector<S>tre;
    Seg(){};
    Seg(int n){
        tre.resize(n<<2);
        build(1,n,1);
    }
    void pushdown(int p){
        if(tre[p].laz!=INF) {
            tre[lc].laz=tre[rc].laz=tre[p].laz;
            tre[lc].laz1=tre[rc].laz1=false;
            tre[lc].mx=tre[rc].mx=tre[p].laz;
            tre[lc].mn=tre[rc].mn=tre[p].laz;
            tre[lc].sum=tre[p].laz*(tre[lc].r-tre[lc].l+1);
            tre[rc].sum=tre[p].laz*(tre[rc].r-tre[rc].l+1);
            tre[p].laz=INF;
        }
        if(tre[p].laz1){
            tre[lc].laz1^=tre[p].laz1;
            tre[rc].laz1^=tre[p].laz1;

            tre[lc].sum=-tre[lc].sum;
            ll lasmn=tre[lc].mn,lasmx=tre[lc].mx;
            tre[lc].mx=-lasmn;
            tre[lc].mn=-lasmx;

            tre[rc].sum=-tre[rc].sum;
            lasmn=tre[rc].mn,lasmx=tre[rc].mx;
            tre[rc].mx=-lasmn;
            tre[rc].mn=-lasmx;
            tre[p].laz1=0;
        }
    }
    void pushup(int p){
        tre[p].mx=max(tre[lc].mx,tre[rc].mx);
        tre[p].mn=min(tre[lc].mn,tre[rc].mn);
        tre[p].sum=tre[lc].sum+tre[rc].sum;
    }
    void build(int l,int r,int p){
        tre[p]={l,r,0,-INF,INF,INF,0};
        if(l==r){
            tre[p].sum=b[l];
            tre[p].mx=tre[p].mn=b[l];
            return;
        }
        int mid=l+r>>1;
        build(l,mid,lc);
        build(mid+1,r,rc); 
        pushup(p);
    }
    void Change(int l,int r,ll k,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            tre[p].laz1=false;
            tre[p].laz=tre[p].mx=tre[p].mn=k;
            tre[p].sum=k*(tre[p].r-tre[p].l+1);
            return;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        if(l<=mid) Change(l,r,k,lc);
        if(r>mid) Change(l,r,k,rc);
        pushup(p);
    }//区间赋值
    void Flip(int l,int r,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            tre[p].laz1^=1;
            tre[p].sum*=-1;
            ll lasmn=tre[p].mn,lasmx=tre[p].mx;
            tre[p].mx=-lasmn;
            tre[p].mn=-lasmx;
            return;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        if(l<=mid) Flip(l,r,lc);
        if(r>mid) Flip(l,r,rc);
        pushup(p);
    }//区间乘-1
    Info qry(int l,int r,int p){
        if(l<=tre[p].l&&tre[p].r<=r){
            return Info(tre[p].sum,tre[p].mx,tre[p].mn);
        }
        int mid=tre[p].l+tre[p].r>>1;
        Info ans;
        pushdown(p);
        if(l<=mid) ans+=qry(l,r,lc);
        if(r>mid) ans+=qry(l,r,rc);
        return ans;
    }
}S;

struct DSU{
    vector<ll>f;
    DSU(){}
    DSU(int n){
        f.resize(n+1);
        iota(f.begin(),f.end(),0);
    };
    int find(int x){
        return x==f[x]?x:f[x]=find(f[x]);
    }
    bool merge(int x,int y){
        int fx=find(x),fy=find(y);
        if(fx==fy) return false;
        f[fx]=fy;
        return true;
    }
};
struct Edge{
    int x,y;ll w;
    Edge(){};
    Edge(int x,int y,ll w):x(x),y(y),w(w){};
    friend bool operator<(Edge A,Edge B){
        return A.w<B.w;
    }
};

        d[u]=d[fa]+1;

        for(int v:G[u]) dfs(dfs,v,u);

int Manacher(const string& st) {
    string s = "^$";
    for (char c : st) {
        s += c;
        s += '$';
    }
    s += '&';
    int len = s.size(), mid = 1, mx = 1, ans = -1;
    vector<int> p(len);
    for (int i = 1; i < len; ++i) {
        if (i < mx)
            p[i] = min(mx - i, p[2 * mid - i]);
        else
            p[i] = 1;
        while (s[i - p[i]] == s[i + p[i]])
            ++p[i];
        if (mx < i + p[i]) {
            mid = i;
            mx = i + p[i];
        }
        ans = max(ans, p[i] - 1);
    }
    return ans;
}

\subsection{数论分块}
\section*{题面简述}

给定一个正整数 $x$，求 $\sum_{i=1}^{x} d(i)$，其中 $d(i)$ 表示 $i$ 的因数个数。


\subsection*{第一步：转化求和式}

由因数个数的定义 $d(i) = \sum_{d \mid i} 1$，将求和交换顺序：
\[
    \sum_{i=1}^{x} d(i) = \sum_{i=1}^{x} \sum_{d \mid i} 1 = \sum_{d=1}^{x} \left\lfloor \frac{x}{d} \right\rfloor
\]

即问题转化为求 $\sum_{d=1}^{x} \left\lfloor \frac{x}{d} \right\rfloor$。

\subsection*{第二步：整除分块}

$\left\lfloor \frac{x}{d} \right\rfloor$ 的取值至多只有 $O(\sqrt{x})$ 种。对于每一种取值 $v = \left\lfloor \frac{x}{l} \right\rfloor$，所有满足 $\left\lfloor \frac{x}{d} \right\rfloor = v$ 的 $d$ 构成一个连续区间 $[l, r]$，其中：
\[
    r = \min\!\left(\left\lfloor \frac{x}{v} \right\rfloor,\; x\right)
\]

在该区间内 $\left\lfloor \frac{x}{d} \right\rfloor$ 恒为 $v$，因此该块的贡献为：
\[
    v \times (r - l + 1)
\]

遍历所有块，累加贡献即得答案：
\[
    \mathrm{Ans} = \sum_{\text{block } [l,r]} \left\lfloor \frac{x}{l} \right\rfloor \times (r - l + 1)
\]

每次令 $l \leftarrow r + 1$ 跳到下一个块，直到 $l > x$。

\section*{复杂度分析}

$\left\lfloor \frac{x}{d} \right\rfloor$ 的不同取值个数为 $O(\sqrt{x})$（当 $d \le \sqrt{x}$ 时至多 $\sqrt{x}$ 种，当 $d > \sqrt{x}$ 时 $\left\lfloor \frac{x}{d} \right\rfloor < \sqrt{x}$，同样至多 $\sqrt{x}$ 种），因此整除分块的块数为 $O(\sqrt{x})$，每块内仅需 $O(1)$ 计算。

总时间复杂度为 $O(\sqrt{x})$，空间复杂度为 $O(1)$。

    ll n;cin>>n;
    ll ans=0;
    for(int l=1,r;l<=n;l=r+1) {
        ll v=(n/l);
        r=min((n/v),n);
        ans+=v*(r-l+1);
    }
    cout<<ans<<endl;

\subsection{莫比乌斯反演}
\section*{题面简述}

给定 $N, M$，求满足 $1 \le x \le N$，$1 \le y \le M$ 且 $\gcd(x, y)$ 为质数的有序对 $(x, y)$ 的个数。

多组询问，$T \le 10^4$，$N, M \le 10^7$。


\subsection*{第一步：枚举质数}

设答案为 $\mathrm{Ans}$，直接按定义展开：
\[
    \mathrm{Ans} = \sum_{p \in \mathcal{P}} \sum_{x=1}^{N} \sum_{y=1}^{M} [\gcd(x, y) = p]
\]
其中 $\mathcal{P}$ 为质数集合。令 $x = pa$，$y = pb$，则：
\[
    \mathrm{Ans} = \sum_{p \in \mathcal{P}} \sum_{a=1}^{\lfloor N/p \rfloor} \sum_{b=1}^{\lfloor M/p \rfloor} [\gcd(a, b) = 1]
\]

\subsection*{第二步：莫比乌斯反演}

利用经典恒等式 $[\gcd(a,b)=1] = \sum_{d \mid \gcd(a,b)} \mu(d)$，代入得：
\[
    \mathrm{Ans} = \sum_{p \in \mathcal{P}} \sum_{d=1}^{\min(\lfloor N/p \rfloor, \lfloor M/p \rfloor)} \mu(d) \left\lfloor \frac{N}{pd} \right\rfloor \left\lfloor \frac{M}{pd} \right\rfloor
\]

\subsection*{第三步：换元合并}

令 $T = pd$，则 $d = T/p$，对于固定的 $T$，$p$ 遍历 $T$ 的所有质因子。交换求和顺序：
\[
    \mathrm{Ans} = \sum_{T=1}^{\min(N,M)} \left\lfloor \frac{N}{T} \right\rfloor \left\lfloor \frac{M}{T} \right\rfloor \sum_{p \mid T,\, p \in \mathcal{P}} \mu\!\left(\frac{T}{p}\right)
\]

定义函数：
\[
    f(T) = \sum_{p \mid T,\, p \in \mathcal{P}} \mu\!\left(\frac{T}{p}\right)
\]

则答案化为：
\[
    \mathrm{Ans} = \sum_{T=1}^{\min(N,M)} f(T) \left\lfloor \frac{N}{T} \right\rfloor \left\lfloor \frac{M}{T} \right\rfloor
\]

\subsection*{第四步：预处理 $f$ 及其前缀和}

$f(T)$ 的预处理方式：线性筛求出所有质数与 $\mu$ 值后，枚举每个质数 $p$，再枚举 $p$ 的倍数 $T = kp$，将 $\mu(k)$ 累加到 $f(T)$ 上。这一步复杂度为：
\[
    \sum_{p \in \mathcal{P}} \left\lfloor \frac{n}{p} \right\rfloor \approx O\!\left(\frac{n \ln \ln n}{1}\right) = O(n \log \log n)
\]

随后求出 $f$ 的前缀和 $F(n) = \sum_{T=1}^{n} f(T)$。

\subsection*{第五步：整除分块回答询问}

对每组询问，利用 $\lfloor N/T \rfloor$ 和 $\lfloor M/T \rfloor$ 的取值分块性质（整除分块/数论分块），将 $T$ 的求和区间划分为 $O(\sqrt{N} + \sqrt{M})$ 个块，每块内 $\lfloor N/T \rfloor$ 和 $\lfloor M/T \rfloor$ 均为常数，利用前缀和 $F$ 即可 $O(1)$ 求出每块的贡献：
\[
    \mathrm{Ans} = \sum_{\text{block } [l,r]} \left\lfloor \frac{N}{l} \right\rfloor \left\lfloor \frac{M}{l} \right\rfloor \bigl(F(r) - F(l-1)\bigr)
\]

\section*{复杂度分析}

\begin{itemize}
    \item \textbf{预处理}：线性筛求 $\mu$ 和质数表为 $O(n)$；枚举质数倍数计算 $f$ 为 $O(n \log \log n)$；前缀和为 $O(n)$。总预处理复杂度 $O(n \log \log n)$，其中 $n = \max(N, M) = 10^7$。
    \item \textbf{单次询问}：整除分块为 $O(\sqrt{n})$。
    \item \textbf{总复杂度}：$O(n \log \log n + T\sqrt{n})$。
\end{itemize}

对于本题 $n = 10^7$，$T = 10^4$：
\begin{itemize}
    \item 预处理：$10^7 \times \log \log(10^7) \approx 10^7 \times 3 \approx 3 \times 10^7$。
    \item 查询：$10^4 \times \sqrt{10^7} \approx 10^4 \times 3162 \approx 3.16 \times 10^7$。
\end{itemize}

总操作量约 $6 \times 10^7$，可以通过。空间复杂度为 $O(n)$。
\begin{lstlisting}[caption={莫比乌斯反演},label={莫比乌斯反演}]
//P2398 GCD-SUM
#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
typedef long long ll;
const int N=1e5+5;
ll p[N],vis[N],cnt;
ll mu[N],pre[N],phi[N];
void init(int n){
    mu[1]=phi[1]=1;
    for(int i=2;i<=n;i++){
        if(!vis[i]){
            p[cnt++]=i;
            mu[i]=-1;
            phi[i]=i-1;
        }
        for(int j=0;i*p[j]<=n;j++){
            int m=i*p[j];
            vis[m]=1;
            if(i%p[j]==0){
                mu[m]=0;
                phi[m]=phi[i]*p[j];
                break;
            }
            phi[m]=phi[i]*(p[j]-1);
            mu[m]=-mu[i];
        }
    }
    for(int i=1;i<=n;i++) pre[i]=pre[i-1]+phi[i];
}
int main(){
    cin.tie(0)->ios::sync_with_stdio(false);
    ll ans=0;
    ll n;cin>>n;
    init(n+5);
    for(int l=1,r;l<=n;l=r+1){
        ll v=n/l;
        r=n/v;
        ans+=v*v*(pre[r]-pre[l-1]);
    }
    cout<<ans<<endl;
    return 0;
}

//P1516青蛙的约会
#include<bits/stdc++.h>
using namespace std;
#define endl '\n'
typedef long long ll;
ll exgcd(ll a,ll b,ll &x,ll &y){
    if(!b){
        x=1,y=0;
        return a;
    }
    ll d=exgcd(b,a%b,y,x);
    y-=a/b*x;
    return d;
}
void solve(){
    int x,y,m,n,l;
    cin>>x>>y>>m>>n>>l;
    ll X,Y;
    ll d=exgcd(n-m,l,X,Y);
    if((x-y)%d) cout<<"Impossible"<<endl;
    else{
        ll s=(x-y)/d;
        ll mod=abs(l/d);
        X*=s;
        Y*=s;
        cout<<(X%mod+mod)%mod<<endl;
    }
}
