struct DSU{
    vector<int>f;
    vector<int>siz;
    DSU(){};
    DSU(int n){
        f.resize(n+1);
        iota(f.begin(),f.end(),0);
        siz.assign(n+1,1);
    }
    int find(int x){
        return x==f[x]?x:f[x]=find(f[x]);
    }
    bool merge(int x,int y){
        int fx=find(x),fy=find(y);
        if(fx==fy) return false;
        f[fx]=fy;
        siz[fy]+=siz[fx];
        return true;
    }
    bool same(int x,int y){
        return find(x)==find(y);
    }
};

struct DSU{
    DSU(){}
    vector<int>f,siz;
    vector<ll>dis;
    DSU(int n){
        dis.resize(n+1);
        f.resize(n+1);
        iota(f.begin(),f.end(),0);
        siz.assign(n+1,1);
    }
    int find(int x){
        int fx=f[x];
        if(fx==x) return x;
        int rt=find(f[x]);
        dis[x]+=dis[f[x]];
        return f[x]=rt;
    }
    bool merge(int x,int y){
        int fx=find(x),fy=find(y);
        if(fx==fy) return false; 
        f[fx]=fy;
        dis[fx]+=siz[fy];
        siz[fy]+=siz[fx];
        return true;
    }
    bool same(int x,int y){
        return find(x)==find(y);
    }
    ll Dis(int x,int y){
        return dis[x]-dis[y];

struct Seg{
    #define lc (p<<1)
    #define rc (p<<1|1)
    struct S{
        int l,r;
        ll val,laz;
    };
    Seg(){};
    Seg(int n){
        tre.resize(n<<2);
        build(0,n,1);
    };
    void pushup(int p){
        tre[p].val=tre[lc].val+tre[rc].val;
    }
    void pushdown(int p){
        if(tre[p].laz==0) return;
        tre[lc].laz+=tre[p].laz;
        tre[rc].laz+=tre[p].laz;
        tre[lc].val+=tre[p].laz;
        tre[rc].val+=tre[p].laz;
        tre[p].laz=0;
    }
    void build(int l,int r,int p){
        tre[p]={l,r,0,0};
        if(l==r){
            tre[p]=a[l];
            return;
        }
        int mid=l+r>>1;
        build(l,mid,lc);
        build(mid+1,r,rc);
        pushup(p);
    }
    void add(int l,int r,ll k,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            tre[p].val+=k*(tre[p].r-tre[p].l+1);
            tre[p].laz+=k;
            return;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        if(l<=mid) add(l,r,k,lc);
        if(r>mid) add(l,r,k,rc);
        pushup(p);
    }
    ll qry(int l,int r,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            return tre[p].val;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        ll ans=0;
        if(l<=mid) ans+=qry(l,r,lc);
        if(r>mid) ans+=qry(l,r,rc);
        return ans;
    }
};

struct Trie{
    vector<int>cnt;
    vector< vector<int> >son;
    int idx=0;
    Trie(int n){
        cnt.resize(n+1);
        son.resize(n+1);
        for(int i=0;i<=n;i++) son[i]=vector<int>(65);
    }
    void insert(string &s){
        int num;
        int p=0;
        for(char &c:s){
            if(isdigit(c)) num=c-'0'+52;
            else if(islower(c)) num=c-'a'+26;
            else if(isupper(c)) num=c-'A';
            if(!son[p][num]) son[p][num]=(++idx);
            p=son[p][num];
            ++cnt[p];
        }
    }
    int query(string &s){
        int num;
        int p=0;
        for(char &c:s){
            if(isdigit(c)) num=c-'0'+52;
            else if(islower(c)) num=c-'a'+26;
            else if(isupper(c)) num=c-'A';
            if(!son[p][num]) return 0;
            p=son[p][num];
        }
        return cnt[p];
    }
}trie;

    int l,r,k;

void Add(int n){
    cnt[a[n]]++;
    res+=cnt[a[n]]*cnt[a[n]]-(cnt[a[n]]-1)*(cnt[a[n]]-1);
}
void Sub(int n){
    cnt[a[n]]--;
    res-=(cnt[a[n]]+1)*(cnt[a[n]]+1)-(cnt[a[n]])*(cnt[a[n]]);
}
int main(){
    ios::sync_with_stdio(false);
    cin.tie(0),cout.tie(0);
    int n,m,k;cin>>n>>m>>k;
    int siz=sqrt(n+0.5);
    for(int i=1;i<=n;i++){
        cin>>a[i];
        pos[i]=i/siz;
    }
    for(int i=0;i<m;i++){
        cin>>q[i].l>>q[i].r;
        q[i].k=i;
    }
    sort(q,q+m,[](Q x,Q y){
        return pos[x.l]==pos[y.l]?x.r<y.r:pos[x.l]<pos[y.l];
    });
    int l=1,r=0;
    for(int i=0;i<m;i++){
        while(q[i].l<l) Add(--l);
        while(q[i].r>r) Add(++r);
        while(q[i].l>l) Sub(l++);
        while(q[i].r<r) Sub(r--);
        ans[q[i].k]=res;
    }
    for(int i=0;i<m;i++){
        cout<<ans[i]<<endl;
    }

const int N=5e5+5;
vector<int>G[N];
int f[N],d[N],dfn[N],top[N],cnt,siz[N],son[N];
void dfs1(int u,int fa){
    d[u]=d[fa]+1;
    f[u]=fa;
    siz[u]=1;
    for(int v:G[u]){
        if(fa==v) continue;
        dfs1(v,u);
        siz[u]+=siz[v];
        if(siz[v]>siz[son[u]]) son[u]=v;
    }
}
void dfs2(int u,int node){
    dfn[u]=(++cnt);//维护dfn序
    top[u]=node;
    if(son[u]) dfs2(son[u],node);
    for(int v:G[u]){
        if(v==f[u]||v==son[u]) continue;
        dfs2(v,v);
    }
}
int LCA(int l,int r){
    while(top[l]!=top[r]){
        if(d[top[l]]<d[top[r]]) swap(l,r);
        l=f[top[l]];
    }
    if(d[l]<d[r]) return l;
    return r;
}//树上LCA跳

    f[u]=fa;
    siz[u]=1;
    for(int v:G[u]){
        if(v==fa) continue;
        dfs1(v,u);
        siz[u]+=siz[v];
        if(siz[v]>siz[son[u]]) son[u]=v;
    }
}
void dfs2(int u,int node){
    top[u]=node;
    dfn[u]=(++cnt);
    b[cnt]=a[u];
    if(son[u]){
        dfs2(son[u],node);
    }
    for(int v:G[u]){
        if(v==f[u]||v==son[u]) continue;
        dfs2(v,v);
    }
}
struct Info{
    ll sum,mx,mn;
    Info(ll sum,ll mx,ll mn):sum(sum),mx(mx),mn(mn){};
    Info():sum(0LL),mx(-INF),mn(INF){};
    friend Info operator+(const Info A,const Info B){
        return Info(A.sum+B.sum,max(A.mx,B.mx),min(A.mn,B.mn));
    }
    friend Info operator+=(Info &A,const Info B){
        A=A+B;
        return A;
    }
};
struct Seg{
    struct S{
        int l,r;
        ll sum,mx,mn,laz;bool laz1;
    };
    #define lc (p<<1)
    #define rc (p<<1|1)
    vector<S>tre;
    Seg(){};
    Seg(int n){
        tre.resize(n<<2);
        build(1,n,1);
    }
    void pushdown(int p){
        if(tre[p].laz!=INF) {
            tre[lc].laz=tre[rc].laz=tre[p].laz;
            tre[lc].laz1=tre[rc].laz1=false;
            tre[lc].mx=tre[rc].mx=tre[p].laz;
            tre[lc].mn=tre[rc].mn=tre[p].laz;
            tre[lc].sum=tre[p].laz*(tre[lc].r-tre[lc].l+1);
            tre[rc].sum=tre[p].laz*(tre[rc].r-tre[rc].l+1);
            tre[p].laz=INF;
        }
        if(tre[p].laz1){
            tre[lc].laz1^=tre[p].laz1;
            tre[rc].laz1^=tre[p].laz1;

            tre[lc].sum=-tre[lc].sum;
            ll lasmn=tre[lc].mn,lasmx=tre[lc].mx;
            tre[lc].mx=-lasmn;
            tre[lc].mn=-lasmx;

            tre[rc].sum=-tre[rc].sum;
            lasmn=tre[rc].mn,lasmx=tre[rc].mx;
            tre[rc].mx=-lasmn;
            tre[rc].mn=-lasmx;
            tre[p].laz1=0;
        }
    }
    void pushup(int p){
        tre[p].mx=max(tre[lc].mx,tre[rc].mx);
        tre[p].mn=min(tre[lc].mn,tre[rc].mn);
        tre[p].sum=tre[lc].sum+tre[rc].sum;
    }
    void build(int l,int r,int p){
        tre[p]={l,r,0,-INF,INF,INF,0};
        if(l==r){
            tre[p].sum=b[l];
            tre[p].mx=tre[p].mn=b[l];
            return;
        }
        int mid=l+r>>1;
        build(l,mid,lc);
        build(mid+1,r,rc); 
        pushup(p);
    }
    void Change(int l,int r,ll k,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            tre[p].laz1=false;
            tre[p].laz=tre[p].mx=tre[p].mn=k;
            tre[p].sum=k*(tre[p].r-tre[p].l+1);
            return;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        if(l<=mid) Change(l,r,k,lc);
        if(r>mid) Change(l,r,k,rc);
        pushup(p);
    }//区间赋值
    void Flip(int l,int r,int p){
        if(l<=tre[p].l&&r>=tre[p].r){
            tre[p].laz1^=1;
            tre[p].sum*=-1;
            ll lasmn=tre[p].mn,lasmx=tre[p].mx;
            tre[p].mx=-lasmn;
            tre[p].mn=-lasmx;
            return;
        }
        int mid=tre[p].l+tre[p].r>>1;
        pushdown(p);
        if(l<=mid) Flip(l,r,lc);
        if(r>mid) Flip(l,r,rc);
        pushup(p);
    }//区间乘-1
    Info qry(int l,int r,int p){
        if(l<=tre[p].l&&tre[p].r<=r){
            return Info(tre[p].sum,tre[p].mx,tre[p].mn);
        }
        int mid=tre[p].l+tre[p].r>>1;
        Info ans;
        pushdown(p);
        if(l<=mid) ans+=qry(l,r,lc);
        if(r>mid) ans+=qry(l,r,rc);
        return ans;
    }
}S;
void Flip(int x,int y){
    while(top[x]!=top[y]){
        if(d[top[x]]<d[top[y]]) swap(x,y);
        S.Flip(dfn[top[x]],dfn[x],1);
        x=f[top[x]];
    }
    if(dfn[x]>dfn[y]) swap(x,y);
    S.Flip(dfn[x]+1,dfn[y],1);
}
Info Qry(int x,int y){
    Info T;
    while(top[x]!=top[y]){
        if(d[top[x]]<d[top[y]]) swap(x,y);
        T+=S.qry(dfn[top[x]],dfn[x],1);
        x=f[top[x]];
    }
    if(dfn[x]>dfn[y]) swap(x,y);
    T+=S.qry(dfn[x]+1,dfn[y],1);
    return T;
}

        return A;
    }
    friend Mat operator*(Mat &A,Mat &B){
        Mat C(A.n);
        for(int k=0;k<=A.n;k++){
            for(int i=0;i<=A.n;i++){
                if(A.a[i][k]==0) continue;
                for(int j=0;j<=A.n;j++){
                    C.a[i][j]=(C.a[i][j]+A.a[i][k]*B.a[k][j]%mod)%mod;
                }
            }
        }
        return C;
    }
    friend Mat operator*=(Mat &A,Mat &B){
        A=A*B;
        return A;
    }
    void q_pow(Mat &A,ll B){
        Mat S(A.n);//单位矩阵
        for(int i=0;i<=A.n;i++) S.a[i][i]=1;
        while(B>0){
            if(B&1){
                S*=A;
            }
            A*=A;
            B>>=1;
        }
        A=move(S);
    }
}M;

const int N=10010;

    dp[1][1]=1;

        for(int j=1;j<=1000;j++){
            dp[i][j]=((j%p)*dp[i-1][j]%p+dp[i-1][j-1])%p;

    }
}
